    def __init__(self, name: str, path_of_results: str):
        self.name = name
        self.path_of_results = path_of_results
        self.figure = None
        self.axs = None
        self.restart_figure()
        self.data = None
        self.initialize()
        self._first_year = datetime.now().date().strftime(FORMAT_TIME)
        return

[docs]    def restart_figure(self, dim: (int, int) = None, fig_size: (int, int) = None) -> None:
        """
        Restart self.figure and self.axis

        :param dim: Dim of grid
        :param fig_size: Size of Figure
        :return: None
        """
        if not dim:
            if fig_size:
                self.figure, self.axs = plt.subplots(figsize=fig_size)
            else:
                self.figure, self.axs = plt.subplots()
        else:
            if fig_size:
                self.figure, self.axs = plt.subplots(*dim, figsize=fig_size)
            else:
                self.figure, self.axs = plt.subplots(*dim)
        return



[docs]    def define_points_of_interest(self, *points_of_interest: [str]) -> None:
        """
        Define points of interest, delete not selected data

        :param points_of_interest: Name of control volumes to add
        :return: None, alter self.points
        """
        to_save = dict()
        for point in points_of_interest:
            to_save[point] = self.data[point]
        self.data = to_save
        return



[docs]    def save_plot(self, file_name: str = None, show: bool = False) -> None:
        """
        Save figure to file_name

        :param file_name: File where save
        :param show: Whether or not show graph
        :return: None, save or show graph generated
        """
        name = file_name
        if not file_name:
            name = "{}.png".format(self.name)
        self.figure.savefig(name)
        if show:
            plt.show()
        return

    def _set_time(self, **kwargs):
        if kwargs.get("in_years", False):
            if kwargs.get("from_year", None):
                if type(kwargs.get("from_year")) is int:
                    self._first_year = "{}-01-01".format(kwargs.get("from_year"))
                else:
                    self._first_year = kwargs.get("from_year")
            modifier = self._to_years
            time_label = "Year"
        else:
            modifier = None
            time_label = "Days"
        return modifier, time_label

    @staticmethod
    def _modify_axes(ax: Axes, data: pd.DataFrame, abscissa: str, ordered: [str], axes_name: (str, str),
                     labels: [str], colors: [str], modifiers: [(callable, callable)] = None,
                     other_args: [dict] = None) -> Axes:
        """
        Modifies an Axes adding plot, names and labels

        :param ax: A plt.Axes to modify
        :param data: Data (a table)
        :param abscissa: Name of column of x axis on data
        :param ordered: Name of column of y axis on data
        :param axes_name: A tuple of x label and y label
        :param labels: Label to add to ordered respectively
        :param colors: Color to add to ordered respectively
        :param modifiers: Functions to apply to a set of data
        :param other_args: Other arguments for plot
        :return: New Axes
        """

        def __identity(an_object: object) -> object:
            return an_object

        x_upper = - float("inf")
        x_downer = float("inf")
        upper_time = None
        downer_time = None
        if other_args is None:
            other_args = [None] * len(ordered)
        if modifiers is None:
            modifiers = [(__identity, __identity)] * len(ordered)
        for index, y in enumerate(ordered):
            if modifiers[index][0] is None:
                modifiers[index] = __identity, modifiers[index][1]
            if modifiers[index][1] is None:
                modifiers[index] = modifiers[index][0], __identity
            try:
                x_downer = min(min(modifiers[index][0](data[abscissa])), x_downer)
                x_upper = max(max(modifiers[index][0](data[abscissa])), x_upper)
            except TypeError:
                # Datetime
                register_matplotlib_converters()
                if upper_time is None:
                    downer_time = min(modifiers[index][0](data[abscissa]))
                    upper_time = max(modifiers[index][0](data[abscissa]))
                else:
                    downer_time = min(min(modifiers[index][0](data[abscissa])), downer_time)
                    upper_time = max(max(modifiers[index][0](data[abscissa])), upper_time)
            if other_args[index] is None:
                ax.plot(
                    modifiers[index][0](data[abscissa]),
                    modifiers[index][1](data[y]),
                    label=labels[index], color=colors[index]
                )
            else:
                ax.plot(
                    modifiers[index][0](data[abscissa]),
                    modifiers[index][1](data[y]),
                    label=labels[index], color=colors[index],
                    **(other_args[index])
                )
        ax.set_xlabel(axes_name[0], fontsize=LABEL_SIZE)
        ax.set_ylabel(axes_name[1], fontsize=LABEL_SIZE)
        if downer_time is None:
            ax.set_xlim(x_downer, x_upper)
        else:
            ax.set_xlim(downer_time, upper_time)
        ax.grid()
        return ax

    # Functions to use on plotting
    def _to_years(self, data: pd.Series) -> pd.Series:
        return pd.to_timedelta(data, unit="d") + pd.to_datetime(self._first_year, format=FORMAT_TIME)

    @staticmethod
    def _minus(data: pd.Series) -> pd.Series:
        return - data
