"""concentration_visualization.py: To plot and show result of concentration calculation"""

import os
import sys
import pandas as pd
import matplotlib.pyplot as plt
from matplotlib.axes import Axes
from molmass import Formula, FormulaError
from matplotlib.figure import Figure
from volume_simulation.visualization import Visualization, TITLE_SIZE, LABEL_SIZE

COLORS = ["b", "r", "maroon", "g", "k", "#ff7f0e"]


def _copy_plot(ax: Axes, new_axes: Axes) -> Axes:
    """
    Copy an axis

    :param ax: Axes to copy
    :param new_axes: New axes to plot ax
    :return: Copied Axes
    """
    lines = list(ax.get_lines())
    for index, line in enumerate(lines):
        new_axes.plot(line.get_data()[0], line.get_data()[1], label=line.get_label(),
                      color=line.get_color())
    new_axes.set_xlabel(ax.get_xlabel(), fontsize=LABEL_SIZE)
    new_axes.set_ylabel(ax.get_ylabel(), fontsize=LABEL_SIZE)
    new_axes.legend(fontsize=LABEL_SIZE)
    new_axes.set_xlim(*ax.get_xlim())
    new_axes.grid()
    return new_axes

[docs]    def initialize(self) -> None:
        """
        Initialize data with all csv on path_of_results

        :return: None, alter self.data
        """
        self.data = dict()
        for file in os.listdir(self.path_of_results):
            if ".csv" in file:
                self.data[file.split(".csv")[0]] = pd.read_csv(
                    os.path.join(self.path_of_results, file)
                )
        return



[docs]    def generate_graph(self, **kwargs) -> Figure:
        """
        Graph of next point of interest

        :param kwargs:
            See below
        :keyword point_of_interest: Control volume to plot
        :keyword elements: List of elements to show
        :keyword phases: List of phases to show
        :keyword concentration_units: Concentration units to show (default mol/L)
        :keyword in_years: If want to plot in years instead of days
        :keyword from_year: year to start
        :keyword limits: a dictionary. Key: element or phases with limit.
            Value: tuple with limits
        :keyword si_on_phases: Add saturation index to phases as a secondary axis
        :keyword show: To show plot on generation
        :return: Return self.figure, and alter it and self.axs
        """
        modifier, time_label = self._set_time(**kwargs)
        to_plot = kwargs.get("point_of_interest")
        temp = self.data[to_plot]
        elements = kwargs.get("elements", [])
        phases = kwargs.get("phases", [])
        plots = len(elements) + len(phases)
        grid_graph = (plots + 1) // 2
        blank = not bool((plots + 1) % 2)
        self.restart_figure((grid_graph, 2), (6 * 2, 4 * grid_graph))
        i, j, c = 0, 0, 0
        species = ""
        for species in elements:

            def species_without_os(element: str) -> str:
                if "(" in element and ")" in element:
                    try:
                        os_candidate = element.split(")")[0].split("(")[1]
                        oxi_state = int(os_candidate)
                        return element.replace("({})".format(oxi_state), "")
                    except ValueError:
                        return element
                else:
                    return element

            def __element(data: pd.Series) -> pd.Series:
                try:
                    return Formula(species_without_os(species)).mass * data
                except FormulaError:
                    return data

            def __m_element(data: pd.Series) -> pd.Series:
                return __element(data) * 1000.0

            units = kwargs.get("concentration_units", "mol/L")
            if species in ["pH", "pe", "temp", "T", "temperature"]:
                units = ""
                func = None
            else:
                func = None
                if units == "g/L":
                    func = __element
                elif units == "mg/L":
                    func = __m_element
                elif units == "mol/L":
                    pass
                else:
                    print("({}) not a unit available. Available ones: g/L, mg/L, mol/L".format(units),
                          file=sys.stderr)
                units = " ({})".format(units)
            try:
                self.axs[i, j] = self._modify_axes(
                    self.axs[i, j], temp, "Days", [species], (time_label, species + units),
                    [species], [COLORS[c]], modifiers=[(modifier, func)]
                )
                self.axs[i, j].legend(fontsize=LABEL_SIZE)
                if kwargs.get("limits", None):
                    self.axs[i, j] = self.__add_limits(kwargs.get("limits"), self.axs[i, j], species)
            except IndexError as e:
                if str(e) == "too many indices for array":
                    self.axs[j] = self._modify_axes(
                        self.axs[j], temp, "Days", [species], (time_label, species + units),
                        [species], [COLORS[c]], modifiers=[(modifier, func)]
                    )
                    self.axs[j].legend(fontsize=LABEL_SIZE)
                    if kwargs.get("limits", None):
                        self.axs[j] = self.__add_limits(kwargs.get("limits"), self.axs[j], species)
                else:
                    raise e
            if j == 1:
                i += 1
            j = (j + 1) % 2
            c = (c + 1) % 6
        for species in phases:
            try:
                self.axs[i, j] = self._modify_axes(
                    self.axs[i, j], temp, "Days", [species], (time_label, "{} (mol/L)".format(species)),
                    [species], [COLORS[c]], modifiers=[(modifier, None)]
                )
                if kwargs.get("limits", None):
                    self.axs[i, j] = self.__add_limits(kwargs.get("limits"), self.axs[i, j], species)
                if kwargs.get("si_on_phases", False) and "si_{}".format(species) in temp.columns:
                    self.__generate_secondary_axis(self.axs[i, j], temp, species, modifier)
                else:
                    self.axs[i, j].legend(fontsize=LABEL_SIZE)
            except IndexError as e:
                if str(e) == "too many indices for array":
                    self.axs[j] = self._modify_axes(
                        self.axs[j], temp, "Days", [species], (time_label, "{} (mol/L)".format(species)),
                        [species], [COLORS[c]], modifiers=[(modifier, None)]
                    )
                    if kwargs.get("limits", None):
                        self.axs[j] = self.__add_limits(kwargs.get("limits"), self.axs[j], species)
                    if kwargs.get("si_on_phases", False) and "si_{}".format(species) in temp.columns:
                        self.__generate_secondary_axis(self.axs[j], temp, species, modifier)
                    else:
                        self.axs[j].legend(fontsize=LABEL_SIZE)
                else:
                    raise e
            if j == 1:
                i += 1
            j = (j + 1) % 2
            c = (c + 1) % 6
        if blank:
            try:
                gs = self.axs[i, 0].get_gridspec()
                self.axs[i, 1].remove()
                big_ax = self.figure.add_subplot(gs[i, 0:])
                _copy_plot(self.axs[i, 0], big_ax)
                self.axs[i, 0].remove()
            except IndexError as e:
                if str(e) == "too many indices for array":
                    gs = self.axs[0].get_gridspec()
                    self.axs[1].remove()
                    big_ax = self.figure.add_subplot(gs[0:])
                    _copy_plot(self.axs[0], big_ax)
                    self.axs[0].remove()
                else:
                    raise e
            if kwargs.get("limits", None):
                self.__add_limits(kwargs.get("limits"), big_ax, species)
            if kwargs.get("si_on_phases", False) and "si_{}".format(species) in temp.columns:
                self.__generate_secondary_axis(big_ax, temp, species, modifier)
            else:
                big_ax.legend(fontsize=LABEL_SIZE)
        self.figure.tight_layout(rect=[0, 0.03, 1, 0.97])
        self.figure.suptitle(to_plot, fontsize=TITLE_SIZE)
        if kwargs.get("show", False):
            plt.show()
        return self.figure


    @staticmethod
    def __add_limits(limits: dict, ax: Axes, element: str) -> Axes:
        """
        Add limits to an axes

        :param limits: Limits (on y axis)
        :param ax: Axes to modify
        :param element: Element to be on limits. If not, axes stay the same
        :return: New axes
        """
        try:
            ax.set_ylim(limits[element])
        except KeyError:
            pass
        return ax

    @staticmethod
    def __generate_secondary_axis(ax: Axes, data: pd.DataFrame,
                                  phase: str, modifier: callable) -> None:
        """
        Generate secondary axis

        :param ax: Axes to modify
        :param data: Data to extract secondary axis
        :param phase: Phase to look for si in data
        :param modifier: Modifier for time (a function)
        :return: None, alter ax given
        """
        if modifier is None:
            def identity(a_data: pd.Series) -> pd.Series:
                return a_data
            modifier = identity
        lines = ax.get_lines()
        si_axis = ax.twinx()
        new_line = si_axis.plot(modifier(data["Days"]), data["si_{}".format(phase)], '.',
                                alpha=0.5,
                                label="Saturation Index", color="gray")
        si_axis.set_ylabel("Saturation Index (mol/L)", fontsize=LABEL_SIZE)
        ax.legend(lines + new_line,
                  [line.get_label() for line in lines] + [new_line[0].get_label()],
                  ncol=2, bbox_to_anchor=(1., -0.15),
                  fontsize=LABEL_SIZE)
        return
