"""volume_visualization.py: To plot and show result of volume calculation"""

import os
import pandas as pd
import matplotlib.pyplot as plt
from matplotlib.figure import Figure
from volume_simulation.visualization import Visualization, TITLE_SIZE, LABEL_SIZE

    def __init__(self, path_of_results: str):
        super().__init__("volume", path_of_results)
        return


[docs]    def initialize(self) -> None:
        """
        Initialize data with all csv on path_of_results

        :return: None, alter self.data
        """
        self.data = dict()
        for folder in os.listdir(self.path_of_results):
            current_path = os.path.join(self.path_of_results, folder)
            if os.path.isdir(current_path):
                self.data[folder] = dict()
                for file in os.listdir(current_path):
                    if ".csv" in file:
                        key = file.rstrip(".csv").split("{}_".format(folder))[-1]
                        self.data[folder][key] = pd.read_csv(
                            os.path.join(current_path, file)
                        )
        return



[docs]    def generate_graph(self, **kwargs) -> Figure:
        """
        Graph of next point of interest

        :param kwargs:
            See below
        :keyword point_of_interest: Control volume to plot
        :keyword in_years: If want to plot in years instead of days
        :keyword from_year: year to start
        :keyword sum_ratios: Add a graph of sum of ratios and its difference from 1.0, default True
        :keyword total_flow: Add a graph of volume in, out and net, default True
        :keyword name of any flow: Name of flows to add, if None, all of them. Argument must be True
        :keyword exclude_flows: No flows to add to graph
        :keyword show: Show figure generated
        :return: Return self.figure, and alter it and self.axs
        """
        modifier, time_label = self._set_time(**kwargs)
        to_plot = kwargs.get("point_of_interest")
        temp_tables = self.data[to_plot]
        table_names = list(temp_tables.keys())
        flows_to_plot = list()
        table_names.remove("volume")
        table_names.remove("mix_ratio")
        for flow in kwargs.keys():
            if flow in [table_name.split("_mix_ratio")[0] for table_name in table_names]:
                flows_to_plot.append("{}_mix_ratio".format(flow))
        if len(flows_to_plot) == 0 and not kwargs.get("exclude_flows", False):
            flows_to_plot = table_names
        bias = 3
        if kwargs.get("sum_ratios", True) and kwargs.get("total_flow", True):
            bias += 1
        grid_graph = (len(flows_to_plot) + bias) // 2
        blank = not bool((len(flows_to_plot) + bias) % 2)
        self.restart_figure((grid_graph, 2), (6 * 2, 4 * grid_graph))
        temp = temp_tables["volume"]
        position_one = 0, 0
        position_two = 0, 1
        if kwargs.get("exclude_flows", False) and not (
                kwargs.get("sum_ratios", True) and kwargs.get("total_flow", True)
        ):
            position_one = 0
            position_two = 1

        if not (kwargs.get("sum_ratios", True) or kwargs.get("total_flow", True)):
            gs = self.axs[position_one].get_gridspec()
            self.axs[position_one].remove()
            self.axs[position_two].remove()
            big_ax = self.figure.add_subplot(gs[0, 0:])
            self._modify_axes(
                big_ax, temp, "Days", ["Volume (m3)"], (time_label, "Volume (m3)"), ["Volume"], ["b"],
                modifiers=[(modifier, None)]
            )
            big_ax.legend(fontsize=LABEL_SIZE)
        else:
            self.axs[position_one] = self._modify_axes(
                self.axs[position_one], temp, "Days", ["Volume (m3)"], (time_label, "Volume (m3)"), ["Volume"], ["b"],
                modifiers=[(modifier, None)]
            )
            self.axs[position_one].legend(fontsize=LABEL_SIZE)

        if kwargs.get("sum_ratios", True):

            def __always_one(data: pd.Series) -> pd.Series:
                return pd.Series([1.0] * len(data))

            temp = temp_tables["mix_ratio"]
            self.axs[position_two] = self._modify_axes(
                self.axs[position_two], temp, "Days", ["Sum Mix Ratios", "Sum Mix Ratios"],
                (time_label, "Ratio"), ["Sum Mix Ratios", "Reference"], ['#ff7f0e', 'b'],
                modifiers=[(modifier, None), (modifier, __always_one)], other_args=[None, {"ls": "--"}]
            )
            self.axs[position_two].legend(fontsize=LABEL_SIZE)

        if kwargs.get("total_flow", True):
            temp = temp_tables["volume"]
            if kwargs.get("sum_ratios", True):
                position_two = 1, 0

            def __net(data: pd.Series) -> pd.Series:
                return data - temp["Volume out (m3)"]

            self.axs[position_two] = self._modify_axes(
                self.axs[position_two], temp, "Days", ["Volume in (m3)", "Volume out (m3)", "Volume in (m3)"],
                (time_label, "Volume (m3)"), ["Volume in", "Volume out", "Net Volume"], ["g", "r", "b"],
                modifiers=[(modifier, None), (modifier, self._minus), (modifier, __net)]
            )
            self.axs[position_two].legend(fontsize=LABEL_SIZE)

        if kwargs.get("sum_ratios", True) and kwargs.get("total_flow", True):
            i, j = 1, 1
        else:
            i, j = 1, 0

        for key in flows_to_plot:
            temp = temp_tables[key]
            label = key.split("_mix_ratio")[0]
            self.axs[i, j] = self._modify_axes(
                self.axs[i, j], temp, "Days", ["Mix Ratio In", "Mix Ratio Out"], (time_label, "Mix Ratio"),
                ["{} In".format(label), "{} Out".format(label)], ['#1f77b4', '#ff7f0e'],
                modifiers=[(modifier, None), (modifier, self._minus)]
            )
            self.axs[i, j].legend(fontsize=LABEL_SIZE)
            if j == 1:
                i += 1
            j = (j + 1) % 2
        if blank:
            self.axs[i, 1].remove()
        self.figure.tight_layout(rect=[0, 0.03, 1, 0.97])
        self.figure.suptitle(to_plot, fontsize=TITLE_SIZE)
        if kwargs.get("show", False):
            plt.show()
        return self.figure
