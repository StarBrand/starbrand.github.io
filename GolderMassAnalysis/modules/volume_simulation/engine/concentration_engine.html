    def __init__(self, path_to_data: str,
                 variable_boundary_conditions: [str] = None,
                 constant_boundary_conditions: [str] = None,
                 constant_boundary_mapper: dict or str = None,
                 *control_volumes: [str], **kwargs):
        super().__init__(variable_boundary_conditions=variable_boundary_conditions,
                         constant_boundary_conditions=constant_boundary_conditions,
                         constant_boundary_mapper=constant_boundary_mapper,
                         log_on_file=kwargs.get("log_on_file", None))
        volume_engine = BasicEngine("flows", "init_volumes.csv", *control_volumes,
                                    variable_boundary_conditions=variable_boundary_conditions,
                                    constant_boundary_conditions=constant_boundary_conditions,
                                    constant_boundary_mapper=constant_boundary_mapper,
                                    logger=self.logger, path=path_to_data)
        self.importer = SystemImporter(volume_engine.system, logger=self.logger)
        file_mapper = mapper_boxes([str(vol) for vol in volume_engine.system.volumes],
                                   [str(vol) for vol in self.system.variable_boundary_conditions],
                                   self.importer.system.external_mapper)
        self.volumes = list(self.importer.volume_mapper.keys())
        if kwargs.get("com", False):
            self.module = IPhreeqcModule(kwargs.get("database", None),
                                         volume_list=self.volumes)
        else:
            self.module = IPhreeqcModuleDLL(kwargs.get("database", None),
                                            volume_list=self.volumes)
        if not kwargs.get("toy", False):
            self._initialize(path_to_data, file_mapper)
        return

    def _initialize(self, path_to_data: str, file_mapper: dict) -> None:
        """
        Add elements, equilibrium and kinetic phases to volumes on
        system (self.importer.system)

        :param path_to_data: Path to data
        :param file_mapper: Where file with initial data are
        :return: None, alter self.importer
        """
        self.logger.debug("Adding initial data")
        for attribute, name in [
            (self.importer.add_elements, "elements"),
            (self.importer.add_equilibrium_phases, "equilibrium_phases"),
            (self.importer.add_kinetic_phases, "kinetic_phases"),
            (self.importer.add_variable_condition, "variable_boundary")
        ]:
            self.logger.debug("\tImporting {} data".format(name))
            attribute(os.path.join(path_to_data, file_mapper[name]["folder"]),
                      file_mapper[name])
        """
        if os.path.exists(os.path.join(path_to_data, "surface.csv")):
            self.logger.debug("\tSurface file found: Importing data")
            self.importer.add_surfaces_from_file(os.path.join(path_to_data, "surface.csv"))
        """
        self.logger.close()
        return

            self.importer.add_header(HEADER_TYPE[header](*args, **kwargs))

[docs]    def run_simulation(self,
                       iterations: int, delta_time: float,
                       **kwargs) -> None:
        """
        Run simulation, put output in a csv file

        :param iterations: Number of iterations
        :param delta_time: Size of time window
        :param kwargs:
            See below
        :keyword path_to_export: Directory for generate "results" folder (default = here)
        :keyword save_template: Path to file where template will be saved (default: None)
        :keyword maximum_lines_in_mem: Maximum lines in memory
        :return: None, generate a file
        """
        path = kwargs.get("path_to_export", "results")
        self.logger.info("Starting calculation")
        start = time.time()
        partial_start = time.time()
        os.makedirs(path, exist_ok=True)
        output = self.step(delta_time, kwargs.get("save_template", None), pd.DataFrame())
        results = dict()
        for vol in self.volumes:
            results[vol] = [
                output.loc[vol].rename(
                    self.importer.system.volumes[0].actual_time
                ).rename(self.importer.synonyms)
            ]
        end = time.time()
        self.logger.info("Day {} completed".format(self.importer.system.volumes[0].actual_time))
        self.logger.debug("\tIt takes {} second".format(end - partial_start))
        saved = False
        # Now iteration!!!
        for current in range(iterations - 1):
            partial_start = time.time()
            try:
                output = self.step(delta_time, kwargs.get("save_template", None), output)
            except Exception as e:
                self.logger.warning("Iteration stop: {}".format(str(e)))
                break
            except KeyboardInterrupt:
                self.logger.warning("Iteration stop interrupted by user")
                break
            for vol in self.volumes:
                try:
                    results[vol].append(output.loc[vol].rename(
                            self.importer.system.volumes[0].actual_time
                    ).rename(self.importer.synonyms))
                except KeyError:
                    results[vol] = [
                        output.loc[vol].rename(
                            self.importer.system.volumes[0].actual_time
                        ).rename(self.importer.synonyms)
                    ]
            end = time.time()
            self.logger.info("Day {} completed".format(self.importer.system.volumes[0].actual_time))
            self.logger.debug("\tIt takes {} second".format(end - partial_start))
            if (current + 1) % kwargs.get("maximum_lines_in_mem", self.MAXIMUM_LINES) == 0:
                results = self.__save_data(saved, self.volumes, results, self.logger, path)
                self.logger.close()
                if not saved:
                    saved = True

        self.logger.info("Writing results on disk")
        self.__save_data(saved, self.volumes, results, self.logger, path)
        del results
        end = time.time()
        self.logger.info("Calculation completed")
        self.logger.info(self.show_time(end, start))
        self.logger.close()
        return



[docs]    def step(self, delta: float, save_template: str or None, updated_data: pd.DataFrame) -> pd.DataFrame:
        """
        Step on iteration

        :param delta: Delta time
        :param save_template: Whether or not save template
        :param updated_data: Data to update parameters
        :return: Result of iteration
        """
        updated_data = updated_data.loc[:, ~updated_data.columns.duplicated()]
        self.importer.step(delta, update=updated_data)
        self.importer.generate()
        if save_template:
            os.makedirs(os.path.join(save_template), exist_ok=True)
            self.importer.template.print_to_file(
                os.path.join(save_template, "Template_day_{}.txt".format(
                    self.importer.system.volumes[0].actual_time
                ))
            )
        return self.module.run_phreeqc(self.importer.template)


    @staticmethod
    def __save_data(append: bool, volumes: [ControlVolume], results: dict, logger: Logger, path: str) -> dict:
        doing = "Writing"
        header = True
        mode = "w"
        if append:
            doing = "Appending"
            header = False
            mode = "a"
        for a_vol in volumes:
            logger.info("\t{} on file {}/{}.csv".format(doing, path, a_vol))
            pd.concat(results[a_vol], axis=1) \
                .rename_axis("Days", axis="columns") \
                .transpose() \
                .to_csv("{}/{}.csv".format(path, a_vol), mode=mode, header=header)
        return dict()
