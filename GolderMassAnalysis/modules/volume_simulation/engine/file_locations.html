[docs]def mapper_boxes(volumes_detected: [str], variable_boundary: [str], constant_boundary: dict) -> dict:
    """
    Generates default location of initial data for all boxes

    :param volumes_detected: Name of control volume boxes generated from
      initial volumes data file (default_location[init_volumes])
    :param variable_boundary: Name of variable boundary condition and file name
    :param constant_boundary: Name of external flow and file name
    :return: Dictionary with location of data to start iterations
    """
    default_location = {
        "init_volumes": "init_volumes.csv",
        "elements": {
            "folder": "initial_conditions"
        },
        "equilibrium_phases": {
            "folder": "equilibrium_phases"
        },
        "kinetic_phases": {
            "folder": "kinetic_phases"
        },
        "variable_boundary": {
            "folder": "variable_boundary_conditions"
        },
        "constant_boundary": {
            "folder": "initial_conditions"
        }
    }
    for vol in volumes_detected:
        default_location["elements"][vol] = "{}.csv".format(vol)
        default_location["equilibrium_phases"][vol] = "{}.csv".format(vol)
        default_location["kinetic_phases"][vol] = "{}.csv".format(vol)
    for boundary in variable_boundary:
        default_location["variable_boundary"][boundary] = "{}.csv".format(boundary)
    for boundary, value in constant_boundary.items():
        default_location["elements"][boundary] = "{}.csv".format(value)
    return default_location
