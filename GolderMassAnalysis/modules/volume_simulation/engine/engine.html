    """
    Engine, runs a simulation of a system
    """
    def __init__(self, variable_boundary_conditions: [str] = None,
                 constant_boundary_conditions: [str] = None,
                 constant_boundary_mapper: dict or str = None,
                 logger: Logger = None,
                 log_on_file: str = None):
        if logger:
            self.logger = logger
        else:
            self.logger = Logger(log_on_file)
            self.logger.info("Running on {}".format(datetime.now().date()))
        self.system = System(variable_boundary_conditions=variable_boundary_conditions,
                             constant_boundary_conditions=constant_boundary_conditions,
                             constant_boundary_mapper=constant_boundary_mapper,
                             logger=self.logger)
        return

[docs]    @staticmethod
    def show_time(end: float, start: float) -> str:
        """
        Show time formatted for logging

        :param end: End time
        :param start: Start time
        :return: Logger of time
        """
        microseconds = end - start - int(end - start)
        microseconds = str(microseconds).split(".")[1]
        running_time = time.gmtime(end - start)
        return """End running on {}\n
        It takes {} hours, {} minutes and {}.{} seconds to complete""".format(
            datetime.now().date(),
            running_time.tm_hour, running_time.tm_min, running_time.tm_sec, microseconds
        )
