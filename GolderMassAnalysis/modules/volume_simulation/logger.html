import os
import sys

[docs]class Logger:
    """
    Generate logging on standard error or file
    """

    def __init__(self, log_on_file: str = None):
        self.file = None
        self.__text = ""
        self.__to_disk = True
        if log_on_file:
            if os.path.exists(log_on_file):
                os.remove(log_on_file)
            self.file = log_on_file
        else:
            self.__to_disk = False
        return


[docs]    def close(self) -> None:
        """
        Close file if it exist

        :return: None, alter self.file
        """
        if self.__to_disk:
            with open(self.file, "a", encoding="utf-8") as file:
                file.write(self.__text)
            self.__text = ""
        return



[docs]    def info(self, message: str) -> None:
        """
        Logging information

        :param message: Message to log
        :return: None, logging or write file
        """
        if self.__to_disk:
            print("INFO:{}".format(message), file=sys.stderr)
            self.__write_file(message, "INFO")
        else:
            logging.info(message)
        return

    def __write_file(self, text: str, level: str) -> None:
        """
        Write on logging file

        :param text: Text to write on self.file
        :param level: Level of logging
        :return: None, logging or write file
        """
        if self.file is None:
            raise RuntimeError("Private method called when file does not exist, this should not happen")
        self.__text += "{}:{}:{}\n".format(level, datetime.now().time(), text)
        return
