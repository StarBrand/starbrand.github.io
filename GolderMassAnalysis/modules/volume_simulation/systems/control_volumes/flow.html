import os
import sys
import numpy as np
import pandas as pd
from volume_simulation import Logger, NullLogger

M3_L = 1e-3
SEC_DAY = 24 * 60 * 60

    """
    Flow class, for flows
    """
    def __init__(self, path: str, control_volumes: (str, str),
                 # Optional parameters
                 logger: Logger = NullLogger(), _toy: bool = False):
        self._inverted = False
        self.times = np.array([])
        self.flow_function = np.array([])
        self.control_volume_1 = control_volumes[0]
        self.control_volume_2 = control_volumes[1]
        self.logger = logger
        if not _toy:
            self._initialize(path)

    def _initialize(self, original_path: str) -> bool:
        """
        Starts reading file

        :param original_path: Original path to look for flow series
        :return: Whether there are no errors
        """
        name = "{}_{}".format(self.control_volume_1, self.control_volume_2)
        path = "{path}/{name}.csv".format(path=original_path, name=name)
        try:
            if not os.path.exists(path):
                name = "{}_{}".format(self.control_volume_2, self.control_volume_1)
                path = "{path}/{name}.csv".format(path=original_path, name=name)
                self._inverted = True
            if not os.path.exists(path):
                raise FileNotFoundError("Not Flow available for this volumes")
            self.logger.debug("Flow {}: Flow initialized using \"{}\" file".format(str(self), path))
            sign = 1
            if self._inverted:
                sign = -1
            data = pd.read_csv(path, sep=",")
            header = data.columns
            time = "Days"
            flow = "Flow"
            for head in header:
                if head in ["Days", "days", "time", "times", "Time"]:
                    time = head
                else:
                    flow = head
            self.flow_function = sign * data[flow]
            self.times = data[time]
            return True
        except BufferError as e:
            print(e, file=sys.stderr)
            self.logger.warning("Flow {}: {}".format(str(self), str(e)))
            return False


[docs]    def get_flows(self, init_time: float, end_time: float, delta_time: float = None) -> np.ndarray:
        """
        Gets an array of volume for a given period of time

        :param init_time: Initial time (days)
        :param end_time: Final time (days)
        :param delta_time: Space between times (days)
        :return: A numpy.array with volumes
        """
        if end_time > max(self.times):
            raise IndexError("Cannot iterate anymore, no more flows")
        if delta_time is None:
            delta_time = 1
        select = np.arange(init_time, end_time + delta_time, delta_time)
        return np.interp(select, self.times, self.flow_function)



[docs]    def get_volumes(self, init_time: float, end_time: float, delta_time: float = None,
                    # Optional parameters
                    time_conversion: float = SEC_DAY, volume_conversion: float = M3_L) -> np.ndarray:
        """
        Get volumes of a time window

        :param init_time: Initial time (days)
        :param end_time: Final time (days)
        :param delta_time: Space between times (days)
        :param time_conversion: (Optional) conversion of time, from seconds
        :param volume_conversion: (Optional) conversion of volume, from liters
        :return: A numpy array of volumes
        """
        if delta_time is None:
            delta_time = 1
        return self.get_flows(init_time, end_time, delta_time) * delta_time * time_conversion * volume_conversion



[docs]    def calculate_mix_ratio(self,
                            volumes: [float] or np.ndarray,
                            init_time: float,
                            delta_time: float = None,
                            # Optional
                            save_to_file: bool = False,
                            append_to_file: bool = False,
                            path_to_file: str = "",
                            time_conversion: float = SEC_DAY,
                            volume_conversion: float = M3_L) -> (np.ndarray, np.ndarray):
        """
        Calculate Mix Ratio of flow

        :param volumes: Volumes calculated
        :param init_time: Initial time for calculate mix ratio
        :param delta_time: Window time for calculation
        :param save_to_file: (Optional) Saves output in a file
        :param append_to_file: (Optional) If append or write from scratch,
            ignore if save_to_file is False
        :param path_to_file: (Optional) path to file, ignore if save_to_file is False
        :param time_conversion: (Optional) conversion of time, from seconds
        :param volume_conversion: (Optional) conversion of volume, from liters
        :return: A numpy array of volumes
        :return: Mix in ratio, Mix out ratio
        """
        if delta_time is None:
            delta_time = 1
        end_time = init_time + (len(volumes) - 1) * delta_time
        if end_time > max(self.times):
            end_time = max(self.times)
        raw_answer = self.get_volumes(init_time, end_time, delta_time, time_conversion, volume_conversion) / volumes
        mix_in, mix_out = np.select([raw_answer > 0], [raw_answer]), np.select([raw_answer <= 0], [-raw_answer])
        if save_to_file:
            if not append_to_file:
                with open(path_to_file,
                          "w", encoding="utf-8") as output:
                    output.write("Days,Mix Ratio In,Mix Ratio Out\n")
            time_vector = np.arange(init_time, end_time + delta_time, delta_time)
            self.logger.debug("Flow {}: Writing file \"{}\"".format(str(self), path_to_file))
            with open(path_to_file, "ab") as p_output:
                np.savetxt(
                    p_output,
                    np.append([time_vector], [mix_in, mix_out], axis=0).transpose(),
                    delimiter=",",
                    newline="\n",
                    fmt=["%.2f", "%.14f", "%.14f"]
                )
        return mix_in, mix_out



[docs]    def invert(self) -> None:
        """
        Inverts flow

        :return: None, alter self to be the opposite flow
        """
        aux = self.control_volume_1
        self.control_volume_1 = self.control_volume_2
        self.control_volume_2 = aux
        self._inverted = not self._inverted
        self.flow_function = - self.flow_function


    def __repr__(self):
        return "{}_{}".format(self.control_volume_1,
                              self.control_volume_2)
