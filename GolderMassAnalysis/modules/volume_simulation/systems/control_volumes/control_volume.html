    def __init__(self, init_volume: float, name: str,
                 max_volume: float = None, min_volume: float = None,
                 logger: Logger = NullLogger()):
        self.max_volume = float("inf")
        self.min_volume = 0.0
        if max_volume is not None:
            self.max_volume = max_volume
        if min_volume is not None:
            self.min_volume = min_volume
        self.actual_time = - 1.0
        self.volume = init_volume
        self.prev_volume = float("nan")
        self.calculated_volumes = np.array([])
        self.calculated_volume = self.volume
        self._name = name
        self.flows = list()
        self.elements = pd.Series()
        self.equilibrium_phases = pd.Series()
        self.kinetic_phases = pd.Series()
        self.saturation_indexes = dict()
        self.logger = logger
        return

[docs]    def add_flow(self, a_flow: Flow, check: bool = True) -> None:
        """
        Add a Flow to volume

        :param a_flow: Flow accordingly
        :param check: (Optional) whether or not check if flow corresponds to volume
        :return: None, alter self.flows
        """
        if check:
            if a_flow.control_volume_1 != self._name:
                if a_flow.control_volume_2 == self._name:
                    print("Inverting Flow", file=sys.stderr)
                    self.logger.warning("Volume {}: Flow inverted for this volume".format(self._name))
                    new_flow = deepcopy(a_flow)
                    new_flow.invert()
                    self._add_flow(new_flow, check)
                    return
                else:
                    raise KeyError("Wrong flow for this volume")
        self._add_flow(a_flow, check)
        return


    def _add_flow(self, a_flow: Flow, check: bool = True) -> None:
        """
        Add a Flow to volume, asserting is not registered

        :param a_flow: Flow accordingly
        :param check: (Optional) whether or not check if flow corresponds to volume
        :return: None, alter self.flows
        """
        if check:
            flows = [str(flow) for flow in self.flows]
            if str(a_flow) in flows:
                self.logger.warning("Volume {}: Flow already saved, updating".format(self._name))
                index = flows.index(str(a_flow))
                self.flows[index] = a_flow
                return
        self.flows.append(a_flow)
        return


[docs]    def volume_balance(self, delta_time: float,
                       # Optional
                       time_conversion: float = SEC_DAY,
                       volume_conversion: float = M3_L
                       ) -> (float, float, float):
        """
        Calculate volume balance

        :param delta_time: Size of time window
        :param time_conversion: (Optional) conversion of time, from seconds
        :param volume_conversion: (Optional) conversion of volume, from liters
        :return: Actual volume
        """
        self.prev_volume = self.volume
        self.calculated_volume = self.volume
        flow_in = 0.0
        flow_out = 0.0
        if self.actual_time < 0:
            self.actual_time = 0.0
        else:
            self.actual_time += delta_time
        for flow in self.flows:
            try:
                flow_value = flow.get_volumes(
                    self.actual_time, self.actual_time, delta_time, time_conversion, volume_conversion
                )[0]
                if flow_value < 0:
                    flow_out -= flow_value
                else:
                    flow_in += flow_value
                self.calculated_volume += flow_value
            except Exception as e:
                raise e
        self.volume = min(self.calculated_volume, self.max_volume)
        self.volume = max(self.volume, self.min_volume)
        return self.volume, flow_in, flow_out



[docs]    def iterate_volume_balance(self, steps: int, delta_time: float,
                               # Optional
                               time_conversion: float = SEC_DAY,
                               volume_conversion: float = M3_L
                               ) -> np.ndarray:
        """
        Iterate volume balance method from self.actual_time to self.actual_time + (steps * delta_time)

        :param steps: Number of iterations
        :param delta_time: Time window
        :param time_conversion: (Optional) conversion of time, from seconds
        :param volume_conversion: (Optional) conversion of volume, from liters
        :return: A 2D numpy array:
            1-dim: Volume of box, in and out
            2-dim: Time (day)
        """
        if self.actual_time < 0:
            self.actual_time = 0.0
        else:
            self.actual_time += delta_time
        init_time = self.actual_time
        if init_time > max(self.flows[0].times):
            raise IndexError("No more flows on file")
        end_time = init_time + ((steps - 1) * delta_time)
        if end_time > max(self.flows[0].times):
            end_time = max(self.flows[0].times)
        self.actual_time = end_time
        flows = list()
        for flow in self.flows:
            flows.append(flow.get_volumes(init_time,
                                          end_time,
                                          delta_time,
                                          time_conversion,
                                          volume_conversion))
        flows = np.array(flows)
        output = np.zeros((3, flows.shape[1]))
        self.calculated_volumes = np.zeros(flows.shape[1])
        self.calculated_volumes[0] = self.volume + flows[:, 0].sum()
        output[0][0] = max(self.calculated_volumes[0], self.min_volume)
        output[0][0] = min(output[0][0], self.max_volume)
        output[1][0] = np.select([flows[:, 0] > 0], [flows[:, 0]]).sum()
        output[2][0] = np.select([flows[:, 0] < 0], [-flows[:, 0]]).sum()
        for i in range(output.shape[1] - 1):
            self.calculated_volumes[i + 1] = output[0][i] + flows[:, i + 1].sum()
            output[0][i + 1] = max(self.calculated_volumes[i + 1], self.min_volume)
            output[0][i + 1] = min(output[0][i + 1], self.max_volume)
            output[1][i + 1] = np.select([flows[:, i + 1] > 0], [flows[:, i + 1]]).sum()
            output[2][i + 1] = np.select([flows[:, i + 1] < 0], [-flows[:, i + 1]]).sum()
        self.prev_volume = output[0][-2]
        self.calculated_volume = self.calculated_volumes[-1]
        self.volume = output[0][-1]
        return output



[docs]    def calculate_mix_ratio(self) -> float:
        """
        Calculates inner mix ratio (Vt-1 / Vt)
        :return: mix ratio
        """
        if isnan(self.prev_volume):
            raise RuntimeError("There is no volume calculated yet")
        return self.prev_volume / self.calculated_volume
