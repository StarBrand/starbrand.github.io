    def __init__(self, variable_boundary_conditions: [str] = None,
                 constant_boundary_conditions: [str] = None,
                 constant_boundary_mapper: dict or str = None,
                 logger: Logger = NullLogger()):
        self.volumes = list()
        self._keys_volumes = dict()
        self.flows = list()
        if variable_boundary_conditions is None:
            self.variable_boundary_conditions = list()
        else:
            self.variable_boundary_conditions = list()
            for boundary in variable_boundary_conditions:
                self.variable_boundary_conditions.append(ExternalFlow(boundary))
        external_name = constant_boundary_mapper
        if constant_boundary_conditions is None:
            constant_boundary_conditions = list()
        if external_name is None:
            external_name = "ext"
        if isinstance(constant_boundary_mapper, dict):
            self.external_mapper = constant_boundary_mapper
        else:
            self.external_mapper = dict()
            for external in constant_boundary_conditions:
                self.external_mapper[external] = external_name
        self.logger = logger
        return

[docs]    def assign_flows(self) -> None:
        """
        Detects and assigns flows to volumes

        :return: None, write self.volumes
        """
        for flow in self.flows:
            try:
                vol_index = self._keys_volumes[flow.control_volume_1]
                self.volumes[vol_index].add_flow(flow, check=True)
                self.logger.info("Flow {} added to {} volume".format(flow, self.volumes[vol_index]))
            except KeyError:
                pass
            try:
                vol_index = self._keys_volumes[flow.control_volume_2]
                other_flow = copy(flow)
                other_flow.invert()
                self.volumes[vol_index].add_flow(other_flow, check=True)
                self.logger.info("Flow {} added to {} volume".format(other_flow, self.volumes[vol_index]))
            except KeyError:
                pass
        return



[docs]    def simulate(self, volume: str, iterations: int, delta_time: float,
                 mix_ratios: bool = False) -> (np.array, np.array):
        """
        Runs simulation, put output in a csv file

        :param volume: Name of volume to return simulation
        :param iterations: Number of iterations
        :param delta_time: Size of time window
        :param mix_ratios: Whether to obtain or not mix ratios
        :return: A tuple:
            Volume in box, flow in and flow out over time
            Mix ratio of volume, flow in and flow out, return an empty array in case it is not solicited
        """
        chosen_one = self.volumes[self._keys_volumes[volume]]
        prev_volume = chosen_one.volume
        self.logger.info("Calculating volume on {}".format(chosen_one))
        result = chosen_one.iterate_volume_balance(iterations, delta_time)
        vol_mix_ratio = np.array([])
        if mix_ratios:
            vol_mix_ratio = result / chosen_one.calculated_volumes
            vol_mix_ratio[0] = np.append(prev_volume, result[0][:-1]) / chosen_one.calculated_volumes
            vol_mix_ratio[2] = - vol_mix_ratio[2]
            vol_mix_ratio = np.append(vol_mix_ratio, [vol_mix_ratio.sum(axis=0)], axis=0)
            vol_mix_ratio[2] = - vol_mix_ratio[2]
        return result, vol_mix_ratio



[docs]    def step(self, delta_time: float) -> pd.DataFrame:
        """
        Generate a new step on computation, returning a pandas DataFrame with
        information per time

        :param delta_time: Window of time (in days)
        :return: Pandas DataFrame:
            Rows: Volume on control volume, self mix ratio, mix ratio in and out per flow
            Columns: Control Volume
        """
        answer = pd.DataFrame({"volume": [], "ratio": []}).transpose()
        for volume in self.volumes:
            inner_answer = dict()
            total, _, _ = volume.volume_balance(delta_time)
            init_time = volume.actual_time
            inner_answer["volume"] = total
            inner_answer["ratio"] = volume.calculate_mix_ratio()
            for flow in volume.flows:
                mix_in, mix_out = flow.calculate_mix_ratio([volume.calculated_volume], init_time, delta_time)
                inner_answer["{}_in".format(flow.control_volume_2)] = mix_in[0]
                inner_answer["{}_out".format(flow.control_volume_2)] = mix_out[0]
            answer = pd.concat([answer, pd.DataFrame({str(volume): inner_answer})], axis=1, sort=False)
        return answer
