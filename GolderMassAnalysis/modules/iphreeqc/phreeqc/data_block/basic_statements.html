ARGUMENTS = {
    "RHO": None,
    "TOT": str,
    "ALK": None,
    "EQUI": str,
    "KIN": str,
    "PARM": int,
    "SI": str
}

    def __init__(self, statement: str, argument: object = None):
        if statement not in ARGUMENTS:
            raise KeyError("Use one of basic statements provided. Eg: BasicStatement.Alkalinity")
        self.statement = statement
        if type(argument) is not ARGUMENTS[self.statement]:
            if argument is ARGUMENTS[self.statement]:
                pass
            else:
                raise ValueError(
                    "Value is not a valid type, valid type {}".format(ARGUMENTS[self.statement])
                )
        self.argument = argument


[docs]    def parse(self, number: int) -> (str, int):
        """
        Parse basic statement

        :param number: Last number used
        :return: Tuple:
            Parsed text
            Number used
        """
        number_used = number + 1
        text = str(number_used * 10)
        text += " PUNCH "
        if self.argument is None:
            text += "{}\n".format(self.statement)
        elif type(self.argument) is str:
            text += "{}(\"{}\")\n".format(self.statement, self.argument)
        else:
            text += "{}({})\n".format(self.statement, self.argument)
        return text, number_used
