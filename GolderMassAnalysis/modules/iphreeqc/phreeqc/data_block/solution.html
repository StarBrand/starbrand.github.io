EXCLUDED_PARAMETERS_ON_PUNCH = ["temperature", "temp", "density", "redox", "units", "pH", "pe"]



[docs]class Solution(DataBlock):
    """
    A solution, view as a chemical source
    """

    __exclusive_parameters = [
        "name", "id", "_info_to_add", "mixture", "_synonyms",
        "surfaces", "equilibrium_phases", "kinetic_phases", "saturation_indexes"
    ]

    def __init__(self, identifier: int, **kwargs):
        super().__init__("SOLUTION")
        self.id = identifier
        self.mixture = Boundary(1)
        self._info_to_add = dict()
        self._synonyms = dict()
        self.surfaces = list()
        self.equilibrium_phases = list()
        self.kinetic_phases = dict()
        self.saturation_indexes = dict()
        for key in kwargs:
            setattr(self, key, kwargs[key])


[docs]    def associates_mixture(self, mixture: Mixture) -> None:
        """
        Associates a mixture to this solution

        :param mixture: A Mixture object
        :return: None, alter self.mixture
        """
        self.mixture = mixture
        self.mixture.associate_solution(
            self.id, self.equilibrium_phases,
            self.kinetic_phases, len(self.surfaces) != 0
        )
        return

[docs]    def add_equilibrium_phase(self, parameter: str or [str]) -> None:
        """
        Add equilibrium phase data

        :param parameter: Name (or list of name) of equilibrium value to add
        :return: None, alter self.equilibrium_phases
        """
        if isinstance(parameter, str):
            self.equilibrium_phases.append(parameter)
        else:
            self.equilibrium_phases.extend(parameter)
        return



[docs]    def add_kinetic_phase(self, parameter: str or [str]) -> None:
        """
        Add kinetic phases data

        :param parameter: Name (or list of name) of kinetic value to add
        :return: None, alter self.kinetic_phases
        """
        if isinstance(parameter, str):
            formula, param = parameter.split("_")
            if self.kinetic_phases.get(formula, None):
                self.kinetic_phases[formula].append(param)
            else:
                self.kinetic_phases[formula] = [param]
        else:
            for param in parameter:
                self.add_kinetic_phase(param)
        return

[docs]    def parse(self, **kwargs) -> str:
        """
        Parse a particular solution in a data block

        :param kwargs:
            See below
        :keyword data: Data to replace values
        :return: Text to be inserted on template
        """
        text = "SOLUTION {}\n".format(self.id)
        if kwargs.get("data", None):
            self.updates_parameters(kwargs.get("data"))
        for attribute in self.__dict__:
            if attribute in Solution.__exclusive_parameters:
                continue
            text += " {} {}".format(attribute, getattr(self, attribute))
            if attribute in self._info_to_add:
                if self._info_to_add[attribute] == "":
                    text += "\n"
                else:
                    text += " {}\n".format(self._info_to_add[attribute])
            else:
                text += "\n"
        eq_phases = True
        for e_phase in self.equilibrium_phases:
            if eq_phases:
                text += "\nEQUILIBRIUM_PHASES {}\n".format(self.id)
                eq_phases = False
            text += " {} {} {}\n".format(e_phase, self.saturation_indexes.get(e_phase, 0),
                                         "{e_phases_%s}" % e_phase)
        k_phases = True
        for k_phase in self.kinetic_phases:
            if k_phases:
                text += "\nKINETICS {}\n".format(self.id)
                k_phases = False
            text += "{}\n".format(k_phase)
            if "formula" not in self.kinetic_phases[k_phase]:
                text += "-formula {}\n".format(k_phase)
            for param in self.kinetic_phases[k_phase]:
                text += "-{}\t{}\n".format(param, "{k_phases_%s_%s}" % (k_phase, param))
        if len(self.surfaces) != 0:
            text += "\n"
            for surface in self.surfaces:
                text += surface.parse(number=self.id)
        text += "END\n\n"
        if kwargs.get("data", None):
            text = text.format_map(SafeDict(**kwargs.get("data")))
        return text



[docs]    def parse_mixture(self, **kwargs) -> str:
        """
        Parse mixture associates with this solution

        :param kwargs: Keyword arguments of mixture.parse()
        :return: DataBlock of mixture
        """
        self._update_user_punch()
        return self.mixture.parse(**kwargs)


    def _update_user_punch(self) -> None:
        """
        Update user punch for values on solution

        :return: None, alter self.mixture.user.punch
        """
        self.mixture.user_punch.add_heading("RHO")
        self.mixture.user_punch.add_punch(BasicStatement.DensityOfSolution)
        for head in self.__dict__:
            if head not in Solution.__exclusive_parameters:
                if head == "Alkalinity":
                    self.mixture.user_punch.add_punch(BasicStatement.Alkalinity)
                    self.mixture.user_punch.add_heading(head)
                elif head in EXCLUDED_PARAMETERS_ON_PUNCH:
                    pass
                else:
                    self.mixture.user_punch.add_punch(BasicStatement.TotalMolality, head)
                    self.mixture.user_punch.add_heading(head)
        return
