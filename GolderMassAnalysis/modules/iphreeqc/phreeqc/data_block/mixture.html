[docs]class Mixture(DataBlock):
    """
    Mixture with elements, equilibrium phases and kinetic phases
    """
    def __init__(self, identifier: int, output: dict = None):
        super().__init__("Mix")
        self.id = identifier
        self.solution = 1
        self.ratio = 1
        self.use = dict()
        self.equilibrium_phases = list()
        self.kinetic_phases = dict()
        self.solutions = dict()
        """
        Output options: defined by default
        """
        self.output = dict()
        if output:
            for optional in output:
                if optional in list(
                        DEFAULT_VALUES["selected_output"]["default"]
                ) + list(DEFAULT_VALUES["selected_output"]["start_up"]):
                    self.output[optional] = output[optional]
                else:
                    pass
        self.user_punch = UserPunch()


[docs]    def associate_solution(self, solution: int, equilibrium_phases: list,
                           kinetic_phases: dict, surfaces: bool) -> None:
        """
        Associate mixture with a solution

        :param solution: An id of solution
        :param equilibrium_phases: Equilibrium phases from solution
        :param kinetic_phases: Kinetic phases from solution
        :param surfaces: If there is surface
        :return: None, alter self.solution and solution
        """
        self.use = dict()
        self.solution = solution
        self.equilibrium_phases = equilibrium_phases
        if len(self.equilibrium_phases) != 0:
            self.use["equilibrium"] = solution
        self.kinetic_phases = kinetic_phases
        if len(self.kinetic_phases) != 0:
            self.use["kinetic"] = solution
        if surfaces:
            self.use["surface"] = solution
        return

[docs]    def parse(self, **kwargs) -> str:
        """
        Parse mixture to add on input template

        :param kwargs:
            See below
        :keyword data: Data to complete numerical values
        :return: Text to add on template
        """
        self._generate_user_punch()
        text = "\n{} {}\n".format(
            self.name,
            self.id
        )
        text += " {} {}\n".format(self.solution, "{ratio}")
        for solution in self.solutions:
            text += " {} {}\n".format(self.solutions[solution], "{%s_in}" % solution)
            text += " {} {}\n".format(self.solution, "{%s_out}" % solution)
        if kwargs.get("data", None):
            text = text.format_map(SafeDict(**kwargs.get("data")))
        text += "\nUSE equilibrium_phases {}\n".format(self.use.get("equilibrium", "none"))
        text += "USE kinetics {}\n".format(self.use.get("kinetic", "none"))
        text += "USE surface {}\n".format(self.use.get("surface", "none"))
        if len(self.output) > 0:
            text += "\nSELECTED_OUTPUT\n"
            for option in self.output:
                if str(self.output[option]) == "True":
                    text += "\t-{} {}\n".format(option, "true")
                elif str(self.output[option]) == "False":
                    text += "\t-{} {}\n".format(option, "false")
                else:
                    text += "\t-{} {}\n".format(option, self.output[option])
        text += self.user_punch.parse()
        text += "END\n"
        return text


    def _generate_user_punch(self) -> None:
        """
        Generate user punch from elements and arguments of Mixture

        :return: None, alter self.punch
        """
        for e_phase in self.equilibrium_phases:
            self.user_punch.add_heading(e_phase)
            self.user_punch.add_punch(BasicStatement.MolesOfEquilibriumPhase, e_phase)
        for ith, k_phase in enumerate(self.kinetic_phases):
            for param in self.kinetic_phases[k_phase]:
                self.user_punch.add_heading("{}_{}".format(k_phase, param))
            self.user_punch.add_punch(BasicStatement.MolesOfKineticReactant, k_phase)
            self.user_punch.add_punch(BasicStatement.ParameterInKinetic, ith + 1)
        for e_phase in self.equilibrium_phases:
            self.user_punch.add_heading("si_{}".format(e_phase))
            self.user_punch.add_punch(BasicStatement.SaturationIndex, e_phase)
        return
