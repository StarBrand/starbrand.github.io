[docs]class Surface(DataBlock):
    """
    Data block that defines the amount and composition of each
    surface in surface assemblage
    """
    def __init__(self, equilibrate: str or int = None, solution_mapper: {str: int} = None,
                 sites_units: str = None, diffuse_layer: float or True = None,
                 cd_music: bool = False, capacitances: (float, float) = None,
                 donnan: dict = None, only_counter_ions: bool = None, no_edl: bool = False,
                 **kwargs):
        super().__init__("SURFACE")
        if equilibrate is None and len(kwargs) == 0:
            raise ValueError("Composition of surface assemblage not defined implicitly neither explicitly")
        if type(equilibrate) is str:
            if solution_mapper is None:
                raise ValueError("A dictionary associating control volume name and "
                                 "solution identifier must be provided")
            else:
                self.equilibrate = solution_mapper[equilibrate]
        else:
            self.equilibrate = equilibrate
        self.surface_binding_site_formula = dict()
        self.surface_binding_site = dict()
        for key, param in kwargs.items():
            try:
                if "name" in param.keys():
                    self.surface_binding_site_formula[key] = param
                else:
                    self.surface_binding_site[key] = param
            except AttributeError:
                pass
        self.sites_units = sites_units
        if self.sites_units not in [None, "absolute", "density"]:
            raise ValueError("Parameter sites units invalid")
        self.diffuse_layer = diffuse_layer
        self.cd_music = cd_music
        self.capacitances = capacitances
        self.Donnan = donnan
        self.only_counter_ions = only_counter_ions
        self.no_edl = no_edl
        return


[docs]    def parse(self, **kwargs) -> str:
        """
        Parse data block in text to use in header

        :param kwargs:
            See below
        :keyword number: Relative number on template
        :return: Text of data block
        """
        text = "{} {}\n".format(self.name, kwargs.get("number"))
        # explicitly
        if self.equilibrate is None:
            for name, to_parse in self.surface_binding_site.items():
                text += self.__parse_surface_binding_site(name, to_parse)
            for name, to_parse in self.surface_binding_site_formula.items():
                text += self.__parse_surface_binding_site_formula(name, to_parse)
        # implicitly
        else:
            text += "\t-equilibrate {}\n".format(self.equilibrate)
            if self.sites_units is not None:
                text += "\t-sites_units\t{}\n".format(self.sites_units)
            if self.cd_music:
                text += "\t-cd_music\n"
            for name, to_parse in self.surface_binding_site.items():
                text += self.__parse_surface_binding_site(name, to_parse)
            for name, to_parse in self.surface_binding_site_formula.items():
                text += self.__parse_surface_binding_site_formula(name, to_parse)
            if self.capacitances is not None:
                text += "\t-capacitances\t{}\t{}\n".format(*self.capacitances)
            if self.Donnan is not None:
                text += self.__parse_donnan(self.Donnan)
            if self.only_counter_ions is not None:
                param = "false"
                if self.only_counter_ions:
                    param = "true"
                text += "\t-only_counter_ions\t{}\n".format(param)
            if self.no_edl:
                text += "\t-no_edl\n"
            if self.diffuse_layer:
                if self.diffuse_layer is True:
                    text += "\t-diffuse_layer"
                else:
                    text += "\t-diffuse_layer\t{}\n".format(self.diffuse_layer)
        return text


    @staticmethod
    def __parse_surface_binding_site(name: str, surface_binding_site: dict) -> str:
        text = "\t{}\t".format(name)
        try:
            text += str(surface_binding_site["sites"])
        except KeyError:
            text += str(surface_binding_site.get("site_density", ""))
        add_text = surface_binding_site.get("specific_area_per_gram", None)
        if add_text:
            text += "\t{}".format(add_text)
        add_text = surface_binding_site.get("grams", None)
        if add_text:
            text += "\t{}".format(add_text)
        dw = surface_binding_site.get("Dw", None)
        if dw:
            text += "\tDw {}".format(dw)
        text += "\n"
        return text

    @staticmethod
    def __parse_surface_binding_site_formula(name: str, surface_binding_site_formula: dict) -> str:
        type_dict = {
            0: "equilibrium_phase",
            1: "kinetic_reactant"
        }
        try:
            type_surface_formula = type_dict[surface_binding_site_formula["type"]]
        except KeyError:
            type_surface_formula = surface_binding_site_formula["type"]
        text = "\t{}\t{}\t{}\t".format(name, surface_binding_site_formula["name"], type_surface_formula)
        add_text = surface_binding_site_formula.get("sites_per_mole", None)
        if add_text:
            text += "{}".format(add_text)
        add_text = surface_binding_site_formula.get("specific_area_per_mole", None)
        if add_text:
            text += "\t{}".format(add_text)
        text += "\n"
        return text

    @staticmethod
    def __parse_donnan(donnan: dict) -> str:
        text = "\t-Donnan"

        def __inner_parse(a_key: str) -> str:
            if a_key in donnan:
                return "\t{} {}".format(a_key, donnan[a_key])
            else:
                return ""

        if "thickness" in donnan:
            text += "\t{}".format(donnan["thickness"])
        text += __inner_parse("debye_lengths")
        text += __inner_parse("limit_ddl")
        text += __inner_parse("viscosity")
        text += "\n"
        return text
