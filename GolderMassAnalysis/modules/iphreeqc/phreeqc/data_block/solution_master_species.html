from iphreeqc.phreeqc.data_block import Header



[docs]class SolutionMasterSpecies(Header):
    """
    Correspondence between element names and
    aqueous primary and secondary species
    """
    def __init__(self, elem_name: str, master_species: str, alkalinity: float,
                 gram_of_element: float, gram_formula_weight: str or float = ""):
        super().__init__("SOLUTION_MASTER_SPECIES")
        self.elem_name = elem_name
        self.master_species = master_species
        self.alkalinity = alkalinity
        self.gram_formula_weight = gram_formula_weight
        self.gram_of_element = gram_of_element


[docs]    def parse(self, **kwargs) -> str:
        """
        Parse solution master species

        :param kwargs:
            See below
        :keyword first: If is he first element in list
        :return: Text of template
        """
        text = super().parse(**kwargs)
        text += "\t{elem_name}\t{master_species}\t{alkalinity}\t{gram_formula_weight}\t{gram_of_element}\n".format(
            elem_name=self.elem_name, master_species=self.master_species,
            alkalinity=self.alkalinity, gram_formula_weight=self.gram_formula_weight,
            gram_of_element=self.gram_of_element
        )
        return text
