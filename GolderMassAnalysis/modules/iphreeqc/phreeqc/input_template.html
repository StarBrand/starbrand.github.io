[docs]    def add_header(self, header: Header) -> None:
        """
        Add header to template

        :param header: A header instance
        :return: None, alter self.headers
        """
        if header.name in self.headers:
            self.headers[header.name].append(header)
        else:
            self.headers[header.name] = [header]
        return



[docs]    def add_solution(self, solution: Solution, id_solution: int = None) -> None:
        """
        Add solution to template

        :param solution: A solution object
        :param id_solution: id of solution, or, assign by default
        :return: None, alter, self.solution
        """
        if id_solution is None:
            available = len(self.solutions) + 1
        else:
            available = id_solution
        solution.id = available
        solution.associates_mixture(solution.mixture)
        self.solutions.append(solution)
        return



[docs]    def encode(self) -> str:
        """
        Encode template in a text

        :return: Text of an input/output template
        """
        text = ""
        for headers in self.headers:
            first = True
            for header in self.headers[headers]:
                text += header.parse(first=first)
                first = False
            text += "\n"
        for solution in self.solutions:
            text += solution.parse(data=self.data.get(solution.id, None))
        for solution in self.solutions:
            text += solution.parse_mixture(data=self.data.get(solution.id, None))
        return text



[docs]    def as_text(self) -> str:
        """
        Encode template in a text

        :return: Text of an input/output template
        """
        return self.encode()
