import numpy as np
import pandas as pd

    def __init__(self, flows: str, init_values: str,
                 variable_boundary_conditions: [str] = None,
                 constant_boundary_conditions: [str] = None,
                 constant_boundary_mapper: dict or str = None,
                 *control_volumes: [str], log_on_file: str = None, path: str = os.getcwd(),
                 logger: Logger = None):
        super().__init__(variable_boundary_conditions=variable_boundary_conditions,
                         constant_boundary_conditions=constant_boundary_conditions,
                         constant_boundary_mapper=constant_boundary_mapper,
                         logger=logger,
                         log_on_file=log_on_file)
        if logger:
            self.logger = logger

        def import_volumes(engine: Engine) -> [str]:
            """
            Imports volumes from path

            :param engine: Self
            :return: List of generated volumes
            """
            volume_values = pd.read_csv("{}/{}".format(path, init_values), sep=",", header=None, index_col=0)
            if len(control_volumes) > 0:
                volumes_to_use = control_volumes
            else:
                volumes_to_use = pd.read_csv("{}/{}".format(path, init_values), header=None)
                volumes_to_use = list(volumes_to_use[0])
            imported = list()
            for volume in volumes_to_use:
                try:
                    engine.system.add_volume(
                        ControlVolume(
                            volume_values[3][volume], volume,
                            max_volume=volume_values[1][volume],
                            min_volume=volume_values[2][volume],
                            logger=self.logger
                        )
                    )
                    imported.append(volume)
                except Exception as e:
                    engine.logger.warning(e)
                    pass
            return imported

        def import_flows(engine: Engine) -> None:
            """
            Imports flows from path

            :param engine: Self
            :return: None, modify self.system
            """
            path_flow = os.path.join(path, flows)
            for flow in os.listdir(path_flow):
                vol1, vol2 = "", ""
                try:
                    vol1, vol2 = flow.split(".")[0].split("_")
                except ValueError:
                    pass
                try:
                    engine.system.add_flow(Flow(path_flow, (vol1, vol2), logger=self.logger))
                except ValueError as e:
                    engine.logger.warning(e)
                except FileNotFoundError as e:
                    engine.logger.warning(e)

        if not os.path.exists(path):
            raise FileNotFoundError("Not a valid path")
        volumes = import_volumes(self)
        list_of_volume = list()
        # Showing volume used
        index = 0
        for index in range(len(volumes) // 7):
            list_of_volume.append("{},".format(", ".join(
                volumes[index * 7:(index + 1) * 7]
            )))
        if len(volumes) % 7 != 0:
            list_of_volume.append(", ".join(
                volumes[(index + 1) * 7:]
            ))
        self.logger.info("Control Volume models to use:\n\t{}".format("\n\t".join(list_of_volume)))
        import_flows(self)
        self.system.assign_flows()
        self.logger.close()
        return


[docs]    def run_simulation(self, iterations: int, delta_time: float, **kwargs) -> None:
        """
        Run simulation, put output in a csv file

        :param iterations: Number of iterations
        :param delta_time: Size of time window
        :param kwargs:
            See below
        :keyword mix_ratios: Whether to obtain or not mix ratios
        :keyword path_to_export: Directory for generate "results" folder (default = here)
        :return: None, generate a file
        """

        def calculation(engine: Engine, system: System, a_volume: ControlVolume, system_time: float) -> None:
            """
            Inner method that do a computation

            :param engine: self
            :param system: Self.system
            :param a_volume: Volume to be computed
            :param system_time: Current time of system,
            :return: Matrix with volumes per time
            """
            with open("{path}/{vol}/{vol}_volumes.csv".format(path=path, vol=a_volume), "ab") as p_output:
                result, ratios_out = system.simulate(str(a_volume), iterations, delta_time, mix_ratios)
                engine.logger.debug("Writing file \"{}\"".format(
                    "{path}/{vol}/{vol}_volumes.csv".format(path=path, vol=a_volume)
                ))
                time_vector = [np.arange(system_time, a_volume.actual_time + delta_time, delta_time)]
                np.savetxt(
                    p_output,
                    np.append(time_vector, result, axis=0).transpose(),
                    delimiter=",",
                    newline="\n",
                    fmt=["%.2f", "%.14f", "%.14f", "%.14f"]
                )
            if mix_ratios:
                engine.logger.debug("Writing file \"{}\"".format(
                    "{path}/{vol}/{vol}_mix_ratio.csv".format(path=path, vol=a_volume)
                ))
                with open("{path}/{vol}/{vol}_mix_ratio.csv".format(path=path, vol=a_volume), "ab") as p_output:
                    np.savetxt(
                        p_output,
                        np.append(time_vector, ratios_out, axis=0).transpose(),
                        delimiter=",",
                        newline="\n",
                        fmt=["%.2f", "%.14f", "%.14f", "%.14f", "%.14f"]
                    )
                for flow in a_volume.flows:
                    flow.calculate_mix_ratio(result[0], system_time, delta_time,
                                             save_to_file=True, append_to_file=system_time >= delta_time,
                                             path_to_file="{}/{}/{}_mix_ratio.csv".format(path, a_volume, flow))
            return

        mix_ratios = kwargs.get("mix_ratios", False)
        path = kwargs.get("path_to_export", "results")
        self.logger.info("Starting calculation")
        start = time.time()
        os.makedirs("{}".format(path), exist_ok=True)
        for volume in self.system.volumes:
            os.makedirs("{}/{}".format(path, volume), exist_ok=True)
            with open("{path}/{vol}/{vol}_volumes.csv".format(path=path, vol=volume),
                      "w", encoding="utf-8") as output:
                output.write("Days,Volume (m3),Volume in (m3),Volume out (m3)\n")
            if mix_ratios:
                with open("{path}/{vol}/{vol}_mix_ratio.csv".format(path=path, vol=volume),
                          "w", encoding="utf-8") as output:
                    output.write("Days,Mix Ratio,Sum Mix Ratios In,Sum Mix Ratios Out,Sum Mix Ratios\n")
        current_time = 0.0
        for volume in self.system.volumes:
            try:
                calculation(self, self.system, volume, current_time)
            except IndexError:
                self.logger.warning("Not more flow registered to go on")
                break
        end = time.time()
        self.logger.info("Calculation completed")
        self.logger.info(self.show_time(end, start))
        self.logger.close()
        return
