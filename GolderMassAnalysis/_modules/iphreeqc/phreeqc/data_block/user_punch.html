from iphreeqc.phreeqc.data_block import DataBlock
from iphreeqc.phreeqc.data_block.basic_statements import BasicStatement

    """
    UserPunch DataBlock. Punch statements

        """
        Add heading to list

        :param heading: A heading or a list of
        :return: None, alter self.heading
        """
        if type(heading) == str:
            self.heading.append(heading)
        else:
            self.heading.extend(heading)
        return



[docs]    def add_punch(self, basic_statement: str, argument: object = None) -> None:
        """
        Add basic program as a basic PUNCH

        :param basic_statement: One of BasicStatement.<provided>
        :param argument: Valid argument, or None in case available
        :return: None, alter self.punch
        """
        self.punch.append(BasicStatement(basic_statement, argument))
        return



[docs]    def parse(self, **kwargs) -> str:
        """
        Parse UserPunch

        :param kwargs:None
        :return: Text on template
        """
        text = "{}\n".format(self.name)
        text += "\t-headings {}\n".format(" ".join(self.heading))
        text += "\t-start\n"
        number = 0
        for punch in self.punch:
            a_punch, number = punch.parse(number)
            text += "\t{}".format(a_punch)
        text += "\t-end\n"
        return text
