[docs]    def add_database(self, *database: [int]) -> None:
        """
        Add database to IPhreeqc

        :param database: Database "code" (int defined on DataBasePhreeqc)
        :return: None, alter self.databases_to_use
        """
        for db_path in database:
            try:
                self.databases_to_use.append(DataBasePhreeqc.path(db_path))
            except IndexError:
                raise AttributeError("Use one of database available. Eg. DataBasePhreeqc.PHREEQC")
        return



[docs]    def run_phreeqc(self, input_template: InputTemplate) -> pd.DataFrame:
        """
        Run phreeqc

        :param input_template: InputTemplate instance with data to run
        :return: DataFrame with solicited data and alter self.template and self.output
        """
        dispatch = Dispatch('IPhreeqcCOM.Object')
        for db_path in self.databases_to_use:
            dispatch.LoadDatabase(db_path)
        try:
            dispatch.RunString(input_template.as_text())
        except pythoncom.com_error as e:
            raise RuntimeError("Phreeqc error: {}".format(e.excepinfo[2]))
        output = dispatch.GetSelectedOutputArray()
        self.output = pd.DataFrame(output[1:], columns=output[0], index=self.__index_names)
        return self.output.copy(deep=True)
