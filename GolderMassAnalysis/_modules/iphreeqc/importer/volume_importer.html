import pandas as pd
from volume_simulation.systems import ControlVolume
from iphreeqc.phreeqc.data_block import Solution, Mixture, Boundary, Surface



[docs]class VolumeImporter:
    """
    Generate a data block from a Control Volume instance
    """
    def __init__(self, volume: ControlVolume, identifier: int):
        self.volume = volume
        self.solution = Solution(identifier)
        self._mixture = self.solution.mixture
        self.id = identifier
        self._data = dict()
        self._output = dict()
        self.surfaces = list()
        return

[docs]    def generate(self, identifier_mixture: int, volume_mapper: dict) -> None:
        """
        Generate solution from volume

        :return: None, alter self.solution
        """
        self._data.clear()
        new_solution = Solution(self.id, **self.volume.elements.to_dict())
        for key, value in self.volume.elements.to_dict().items():
            self._data[key] = value
        for surface in self.surfaces:
            new_solution.add_surface(surface)
        if identifier_mixture > 0:
            mixture = Mixture(identifier_mixture, output=self._output)
        else:
            mixture = Boundary(identifier_mixture)
        for flow in self.volume.flows:
            try:
                flow_id = volume_mapper[flow.control_volume_2]
                mixture.add_solution(flow_id, flow.control_volume_2)
            except KeyError as e:
                raise e
        if len(self.volume.equilibrium_phases) > 0:
            for key, value in self.volume.equilibrium_phases.to_dict().items():
                self._data["e_phases_{}".format(key)] = value
                new_solution.add_equilibrium_phase(key)
        if len(self.volume.kinetic_phases) > 0:
            for key, value in self.volume.kinetic_phases.to_dict().items():
                self._data["k_phases_{}".format(key)] = value
                new_solution.add_kinetic_phase(key)
        if len(self.volume.saturation_indexes) > 0:
            for key, value in self.volume.saturation_indexes.items():
                new_solution.add_saturation_index(key, value)
        self.solution = new_solution
        self.solution.associates_mixture(mixture)
        self._mixture = mixture
        return

[docs]    def add_data(self, step_result: pd.DataFrame) -> None:
        """
        Add Data from result of a step iteration

        :param step_result: Result of a system.step()
        :return: None, alter self._data
        """
        new_data = step_result[str(self.volume)].to_dict()
        for key in new_data:
            self._data[key] = new_data[key]
        return
