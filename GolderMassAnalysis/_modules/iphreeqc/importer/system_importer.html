    def __init__(self, system: System, logger: Logger = NullLogger(), _toy: bool = False):
        self.counters = {
            "solutions": 1,
            "mixtures": 1
        }
        self.system = system
        self.volume_importers = list()
        self.variable_boundary_importer = list()
        self.__variable_chemical = dict()
        self._solution_mapper = dict()
        self.logger = logger
        self.external_importers = list()
        self.dependent_importers = list()
        self.template = InputTemplate()
        self._add_info_to = dict()
        self.synonyms = dict()
        self._static_elements = dict()
        self.volume_mapper = dict()

        if not _toy:
            solutions = self.counters["solutions"]
            mix_index = self.counters["mixtures"]
            self.__external_solutions = dict()
            for volume in self.system.volumes:
                self.volume_mapper[str(volume)] = mix_index
                self._solution_mapper[str(volume)] = solutions
                self.volume_importers.append(VolumeImporter(volume, solutions))
                solutions += 1
                mix_index += 1
            for boundary in system.variable_boundary_conditions:
                self.variable_boundary_importer.append(VolumeImporter(boundary, solutions))
                self._solution_mapper[str(boundary)] = solutions
                solutions += 1
            for volume in self.volume_importers:
                self.logger.debug("Working with volume {}".format(str(volume.volume)))

                def _generate_volume(a_system: SystemImporter,
                                     a_volume: VolumeImporter,
                                     solution_index: int) -> int:
                    try:
                        a_volume.generate(a_system.volume_mapper[str(a_volume.volume)],
                                          a_system._solution_mapper)
                        return solution_index
                    except KeyError as e:
                        a_system.logger.debug("Adding external control volumes {}".format(e.args[0]))
                        if e.args[0] in a_system.system.external_mapper.keys():
                            external = a_system.system.external_mapper[e.args[0]]
                        else:
                            external = "ext"
                        try:
                            sol_external = a_system.__external_solutions[external]
                            a_system._solution_mapper[e.args[0]] = sol_external
                        except KeyError:
                            a_system.__external_solutions[external] = solution_index
                            a_system._solution_mapper[e.args[0]] = solution_index
                            a_system.system.external_mapper[external] = external
                            solution_index += 1
                        return _generate_volume(a_system, a_volume, solution_index)

                solutions = _generate_volume(self, volume, solutions)
            for boundary in self.variable_boundary_importer:
                boundary.generate(-1, self._solution_mapper)
            for ext, sol_ext in self.__external_solutions.items():
                if "->" in ext:
                    self.external_importers.append(VolumeImporter(ExternalFlow(ext.split("->")[0]), sol_ext))
                else:
                    self.external_importers.append(VolumeImporter(ExternalFlow(ext), sol_ext))
            for index, _ in enumerate(self.external_importers):
                self.external_importers[index].generate(-1, self._solution_mapper)
            self.counters["solutions"] = solutions
            self.counters["mixture"] = mix_index
            self.logger.info("Solutions: {}".format(self._solution_mapper))
            self.logger.info("Mixtures: {}".format(self.volume_mapper))

[docs]    def add_output(self, output: str, value: object, *control_volumes) -> None:
        """
        Add output to all volumes (or particular one(s))

        :param output: Name of selected output option to add
        :param value: Value of selected output
        :param control_volumes: Control volumes to add selected output
        :return: None, alter [solution.mixture for solution in self.template.solutions]
        """
        if control_volumes:
            for volume in control_volumes:
                for index, _ in enumerate(self.volume_importers):
                    if self.volume_importers[index].id == self.volume_mapper[str(volume)]:
                        self.volume_importers[index].add_output(output, value)
        else:
            for index, _ in enumerate(self.volume_importers):
                self.volume_importers[index].add_output(output, value)
        return



[docs]    def add_surface_from_data(self, data: dict = None, to_all: bool = False) -> None:
        """
        Add surface to input template from data

        :param data: Two options:
            Dictionary: if to_all=True, add surface to all solutions with data on this dict
            Dict of dict: In case to_all=False, this is a dictionary containing control:
                key: Control volume name
                value: data as described before
        :param to_all: Add to all control volume
        :return: None, alter self.template
        """

        def adding_surface(a_data: dict) -> dict:
            building_kwargs = dict()
            for arbitrary_name, params in a_data.items():
                building_kwargs[arbitrary_name] = params
            return building_kwargs

        if to_all:
            to_enter = adding_surface(data)
            for volume in self.volume_importers:
                volume.add_surface(Surface(**to_enter))
        else:
            for name, particular_data in data.items():
                to_enter = adding_surface(particular_data)
                volumes = [str(vol.volume) for vol in self.volume_importers]
                self.volume_importers[volumes.index(name)].add_surface(Surface(**to_enter))
        return



[docs]    def add_elements(self, path: str, mapper: dict = None) -> None:
        """
        Add elements for folder (in path). Search for file with name
        on mapper

        :param path: Path to file or folder
        :param mapper: A dictionary indicating where file is
        :return: None, alter self.volume_importers
        """
        if mapper:
            for boundary in self.variable_boundary_importer:
                try:
                    file = mapper[str(boundary.volume)]
                    boundary.volume.add_elements(os.path.join(path, file))
                except KeyError:
                    pass
            for volume in self.volume_importers:
                file = mapper[str(volume.volume)]
                volume.volume.add_elements(os.path.join(path, file))
            for index, _ in enumerate(self.external_importers):
                try:
                    file = mapper[str(self.external_importers[index].volume)]
                    self.external_importers[index].volume.add_elements(os.path.join(path, file))
                except (KeyError, AttributeError):
                    pass
        else:
            for boundary in self.variable_boundary_importer:
                boundary.volume.add_elements(path)
            for volume in self.volume_importers:
                volume.volume.add_elements(path)
            for external in self.external_importers:
                external.volume.add_elements(path)
        for boundary in self.variable_boundary_importer:
            boundary.generate(-1, self._solution_mapper)
        for external in self.external_importers:
            external.generate(-1, self._solution_mapper)
        for volume in self.volume_importers:
            volume.generate(self.volume_mapper[str(volume.volume)],
                            self._solution_mapper)
        return



[docs]    def add_variable_condition(self, path: str, mapper: dict = None) -> None:
        """
        Add variable chemical attributes with a pandas DataFrame
        Chemical conditions over time are taken from file

        :param path: Path to file or folder
        :param mapper: A dictionary indicating where file is
        :return: None, alter self.__tailings_chemical
        """
        if mapper:
            for boundary in self.variable_boundary_importer:
                file = mapper[str(boundary.volume)]
                self.__variable_chemical[str(boundary.volume)] = pd.read_csv(
                    os.path.join(path, file)
                )
        else:
            for boundary in self.variable_boundary_importer:
                self.__variable_chemical[str(boundary.volume)] = pd.read_csv(path)
        for boundary in self.variable_boundary_importer:
            if str(boundary.volume) in self.__variable_chemical.keys():
                data = self.__variable_chemical[str(boundary.volume)]
                boundary.volume.elements = data[data.Days == min(data.Days)].drop("Days", axis=1).iloc[-1]
                boundary.generate(-1, self._solution_mapper)
        return

[docs]    def add_saturation_index(self, parameter: str, saturation_index: float, *control_volumes: [str]) -> None:
        """
        Add saturation index to equilibrium phases

        :param parameter: Parameter to have si (different from 0)
        :param saturation_index: A valid SI
        :param control_volumes: (Optional) specifies control volumes to add Saturation
            Index different from zero
        :return: None, alter self.volume_importers
        """
        if len(control_volumes) <= 0:
            for volume in self.volume_importers:
                volume.volume.add_saturation_index(parameter, saturation_index)
        else:
            for volume in self.volume_importers:
                if str(volume.volume) in control_volumes:
                    volume.volume.add_saturation_index(parameter, saturation_index)
        return



[docs]    def generate(self) -> None:
        """
        Generate the InputTemplate

        :return: None, alter self.template
        """
        self.template.solutions.clear()
        for volume in self.volume_importers:
            self.template.data[self._solution_mapper[str(volume.volume)]] = volume.get_data()
            self.template.add_solution(volume.solution, self._solution_mapper[str(volume.volume)])
        for boundary in self.variable_boundary_importer:
            self.template.data[self._solution_mapper[str(boundary.volume)]] = boundary.get_data()
            self.template.add_solution(boundary.solution, self._solution_mapper[str(boundary.volume)])
        for external in self.external_importers:
            self.template.data[external.id] = external.get_data()
            self.template.add_solution(external.solution, external.id)
        for index, _ in enumerate(self.template.solutions):
            for parameter, value in self._add_info_to.items():
                self.template.solutions[index].add_info_to(parameter, value)
            for parameter, value in self.synonyms.items():
                self.template.solutions[index].element_as(parameter, value)
            for parameter, value in self._static_elements.items():
                self.template.solutions[index].update_parameter(parameter, value)
        return



[docs]    def step(self, delta_time: float, update: pd.DataFrame = pd.DataFrame()) -> None:
        """
        Generate a step on system

        :param delta_time: Time window size, in days
        :param update: A pandas DataFrame with concentration results
            of previous iteration. Index must be the assigned name
            of control volumes
        :return: None, alter self.system
        """
        step_result = self.system.step(delta_time)
        for volume in self.volume_importers:
            if not update.empty:
                volume.volume.elements.update(update.loc[str(volume.volume)])
                volume.volume.equilibrium_phases.update(update.loc[str(volume.volume)])
                volume.volume.kinetic_phases.update(update.loc[str(volume.volume)])
            volume.generate(self.volume_mapper[str(volume.volume)],
                            self._solution_mapper)
            volume.add_data(step_result)
        for boundary in self.variable_boundary_importer:
            if str(boundary.volume) in self.__variable_chemical.keys():
                data = self.__variable_chemical[str(boundary.volume)]
                actual_time = self.system.volumes[0].actual_time
                boundary.volume.elements.update(
                    pd.Series(
                        [np.interp(actual_time, data["Days"], data[elem]) for elem in data.columns[1:]],
                        index=data.columns[1:]
                    )
                )
            boundary.generate(-1, self._solution_mapper)
        return
